INSERT INTO hasindatabase.BakeryTBL (`PCNumber`, `Date`, `ItemName`, `Price`, `UnitSold`, `SoldAmount`)
SELECT `PC Number`, `Date`, `Item`, `Price`, `Items Sold`, `Sold Amount`
FROM hasindatabase.TempTable
Where Item like "%donut%" OR
Item like "%upcharge%" OR
Item like "%fritter%" OR
Item like "%Coffee Roll%" OR
Item like "%munchkins%" OR
Item like "1 muffin%" OR
Item like "12 muffin%" OR
Item like "4 muffin%" OR
Item like "6 muffin%"
Order By Item

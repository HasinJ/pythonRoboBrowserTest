INSERT INTO ColdBeverageTBL (`PCNumber`, `Date`, `ItemName`, `Price`, `UnitSold`, `SoldAmount`)
SELECT `PC Number`, `Date`, `Item`, `Price`, `Items Sold`, `Sold Amount`
FROM TempTable
Where Item like "%cold brew%" OR
Item like "%iced coffee%" OR
Item like "%iced americano%" OR
Item like "%iced Cappuccino%" OR
Item like "%iced chai%" OR
Item like "%iced latte%" OR
Item like "%Iced Macchiato%" OR
Item like "%Iced Matcha%" OR
Item like "%Iced Pink%" OR
Item like "%Iced Tea%" OR
Item like "%Iced Signature%" OR
Item like "%Cup of Water%"
Order By Item
